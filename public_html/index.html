Using the hash function by Nodejs Crypto

After signing up, the user can log in to their account using their 'email' and 'password'. Then in the index.js file, the checking process will happen as follows: 

let email = hash(inemail);
let password = hash(inpassword);

db.get(query, [email], (err, row) => {
    if (err) {
        return next(err);
    }
    if (!row) {
        // No user found with the provided email
        return res.status(401).send({ error: 'Invalid email or password' });
    }
    if (password === row.password) {
        // User successfully logged in
        res.send('Login successful');
    } else {
        res.status(401).send({ error: 'Invalid email or password' });
    }
});

If the hash of the user's password matches the one saved in the database, the user successfully logs in.
